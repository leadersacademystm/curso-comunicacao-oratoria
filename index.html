  <script>

    window.addEventListener('DOMContentLoaded', () => {

      const questions=[

        {title:"Quando estou diante de uma plateia, meu maior medo é…",options:[{key:'A',label:'Não convencer ou influenciar'},{key:'B',label:'Não gerar uma conexão emocional'},{key:'C',label:'Que minha explicação seja confusa'},{key:'D',label:'Parecer distante ou inacessível'},{key:'E',label:'Questionarem meus dados ou credibilidade'}]},

        {title:"Durante minha fala, eu costumo…",options:[{key:'A',label:'Focar em argumentos que levam a uma decisão'},{key:'B',label:'Usar histórias e apelo emocional para motivar'},{key:'C',label:'Estruturar o conteúdo em etapas claras, como uma aula'},{key:'D',label:'Interagir e criar um ambiente de diálogo'},{key:'E',label:'Apresentar fatos e evidências para embasar minha fala'}]},

        {title:"Para prender a atenção, confio no meu…",options:[{key:'A',label:'Poder de persuasão e na força dos argumentos'},{key:'B',label:'Energia, paixão e na capacidade de inspirar'},{key:'C',label:'Organização, didática e na clareza da explicação'},{key:'D',label:'Empatia, autenticidade e na conexão com as pessoas'},{key:'E',label:'Conhecimento técnico e na minha credibilidade'}]},

        {title:"Me vejo mais como alguém que é…",options:[{key:'A',label:'Persuasivo e orientado a resultados'},{key:'B',label:'Motivador e carismático'},{key:'C',label:'Didático e organizado'},{key:'D',label:'Acessível e empático'},{key:'E',label:'Técnico e especialista no assunto'}]}

      ];

      const keyToProfile={

        A:{name:'Vendedor',desc:'Persuasivo e focado em resultados. Sua força está em convencer e influenciar decisões.'},

        B:{name:'Inspirador',desc:'Emocional, motivacional e carismático. Você se destaca por conectar emocionalmente e motivar a ação.'},

        C:{name:'Instrutor',desc:'Didático, organizado e claro. Sua principal habilidade é transmitir conhecimento eficazmente.'},

        D:{name:'Conector',desc:'Empático, próximo e acessível. Você tem a capacidade de criar relacionamentos genuínos.'},

        E:{name:'Autoridade',desc:'Técnico, respeitado e confiável. Sua credibilidade e expertise são seus maiores trunfos.'}

      };

      let current=0,answers=[];

      const qIndexEl=document.getElementById('qIndex'),barEl=document.getElementById('bar'),qContainer=document.getElementById('qContainer');

      const startBtn=document.getElementById('startBtn'),nextBtn=document.getElementById('nextBtn'),prevBtn=document.getElementById('prevBtn'),resetBtn=document.getElementById('resetBtn');

      function goTo(step){['step-welcome','step-questions','step-processing','step-result'].forEach((id,idx)=>document.getElementById(id).classList.toggle('active',idx===step));if(step===1){current=0;answers=[];renderQ();updateProgress();}}

      function renderQ(){const q=questions[current];qIndexEl.textContent=current+1;let html=`<h2 style=\"margin:8px 0 16px 0; color:var(--primary); font-size: 1.6rem; line-height: 1.3;\">${q.title}</h2><fieldset class=\"options\">`;q.options.forEach(opt=>html+=`<label class='opt'><input type='radio' name='q${current}' value='${opt.key}'/><span><strong>${opt.key})</strong> ${opt.label}</span></label>`);html+='</fieldset>';qContainer.innerHTML=html;}

      function updateProgress(){barEl.style.width=((current)/questions.length*100)+'%';}

      function nextQ(){const sel=document.querySelector(`input[name='q${current}']:checked`);if(!sel){alert('Selecione uma alternativa para continuar.');return;}answers[current]=sel.value;if(current<questions.length-1){current++;renderQ();updateProgress();}else{barEl.style.width='100%';processResult();}}

      function prevQ(){if(current===0)goTo(0);else{current--;renderQ();updateProgress();}}

      function processResult(){goTo(2);const msgs=['Analisando suas respostas…','Calculando seu perfil de comunicador…','Concluído.'];let i=0;const pm=document.getElementById('procMsg');(function tick(){pm.textContent=msgs[i++];if(i<msgs.length)setTimeout(tick,900);else setTimeout(showResult,600);})();}

      function showResult(){const counts={A:0,B:0,C:0,D:0,E:0};answers.forEach(k=>counts[k]++);const sorted=Object.entries(counts).sort((a,b)=>b[1]-a[1]),top=sorted[0],second=sorted[1];const resTitle=document.getElementById('resTitle'),resSubtitle=document.getElementById('resSubtitle'),resPill=document.getElementById('resPill');if(top[1]===second[1]&&top[1]!==0){const p1=keyToProfile[top[0]].name,p2=keyToProfile[second[0]].name;resTitle.textContent=`${p1} + ${p2}`;resSubtitle.textContent=`Você transita entre os perfis ${p1} e ${p2}, unindo o melhor dos dois mundos.`;resPill.textContent=`Empate: ${top[1]}×${second[1]}`;}else{const p=keyToProfile[top[0]];resTitle.textContent=p.name;resSubtitle.textContent=p.desc;resPill.textContent=`Predominante: ${top[1]} ponto(s)`;}goTo(3);}

      startBtn.addEventListener('click',()=>goTo(1));

      nextBtn.addEventListener('click',nextQ);

      prevBtn.addEventListener('click',prevQ);

      resetBtn.addEventListener('click',()=>goTo(0));

    });

  </script>
